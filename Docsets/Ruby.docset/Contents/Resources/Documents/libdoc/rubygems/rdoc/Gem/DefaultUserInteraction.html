Gem::DefaultUserInteraction

Module that defines the default UserInteraction. Any class including this module will have access to the ui method that returns the default UI.

def self.use_ui(new_ui)
  old_ui = @ui
  @ui = new_ui
  yield
ensure
  @ui = old_ui
end
